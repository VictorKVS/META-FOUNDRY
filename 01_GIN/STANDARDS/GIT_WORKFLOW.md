## Python-—Å–∫—Ä–∏–ø—Ç AutoDoc Generator (generate_appendices.py)

–ü—Å–µ–≤–¥–æ–∫–æ–¥ –∑–∞–º–µ–Ω—ë–Ω –ø–æ–ª–Ω–æ—Å—Ç—å—é —Ä–∞–±–æ—á–∏–º —à–∞–±–ª–æ–Ω–æ–º.

```
import yaml
import os
from pathlib import Path

CONFIG_PATH = "autodoc/config/autodoc.yaml"


def load_config():
    with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
        return yaml.safe_load(f)


def ensure_file(path, template=""):
    if not Path(path).exists():
        with open(path, "w", encoding="utf-8") as f:
            f.write(template)
            print(f"[+] Created: {path}")


def update_appendices(cfg):
    for app in cfg.get("appendices", []):
        ensure_file(app["file"], f"# {app['name']}

(–°–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω–æ AutoDoc)
")


def update_toc(document_path):
    with open(document_path, 'r', encoding='utf-8') as f:
        lines = f.readlines()

    toc = ["## –û–≥–ª–∞–≤–ª–µ–Ω–∏–µ
"]
    for line in lines:
        if line.startswith("## "):
            title = line[3:].strip()
            anchor = title.lower().replace(" ", "-")
            toc.append(f"- [{title}](#{anchor})
")

    return toc


def apply_toc(cfg):
    for doc in cfg.get("documents", []):
        if not doc.get("toc", False):
            continue

        path = doc["file"]
        toc = update_toc(path)

        with open(path, 'r', encoding='utf-8') as f:
            content = f.readlines()

        with open(path, 'w', encoding='utf-8') as f:
            f.write("".join(toc) + "
---
" + "".join(content))

        print(f"[+] TOC updated in {path}")


def main():
    cfg = load_config()
    update_appendices(cfg)
    apply_toc(cfg)


if __name__ == "__main__":
    main()
```
